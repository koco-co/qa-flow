private void checkColumnTypeAndSetting(FunctionBO functionBO, Map<String, String> columnNameTypeMap,
                                       List<MultiTableSettingDTO> verifySettingDTOList){
    boolean empty = true;
    for (String columnName : functionBO.getColumnNameList()) {
        String columnType = columnNameTypeMap.get(columnName);
        boolean isNumberType = DataType.SUPPORT_NUMBER_COLUMN_TYPE.contains(columnType.toUpperCase());
        boolean isStringType = DataType.SUPPORT_STRING_COLUMN_TYPE.contains(columnType.toUpperCase());
        if (!isNumberType && !isStringType){
            continue;
        }
        for (MultiTableSettingDTO settingDTO : verifySettingDTOList) {
            Boolean settingIsNumber = MultiTableSettingEnum.isNumber(settingDTO.getKey());
            if (settingIsNumber.equals(isNumberType) || !settingIsNumber.equals(isStringType)){
                empty = false;
                break;
            }
        }
    }
    if (CollectionUtils.isNotEmpty(functionBO.getColumnNameList()) && empty){
        throw new BizException("一致性比对规则，配置的字段没有符合的规则");
    }
}

问题分析：

这段前置校验把“有校验字段但比对细节为空”直接当成非法输入，而没有把它视为默认等值比对。异常不是因为缺少实现，而是校验顺序提前把合法场景拦掉了。

环境证据： 同一个类下方已经实现了 verifySettingDTO 为空时的默认等值比对，说明根因是前置校验与默认分支互相冲突。
